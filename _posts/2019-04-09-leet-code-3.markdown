class Solution {
    func maxSubArray(_ nums: [Int]) -> Int {
        var max = nums[0]

        (0...(nums.count-1)).forEach { (n) in
            (0...n).forEach({ (m) in
            let sum = nums[m...n].reduce(0, +)
                if sum > max {
                    max = sum
                    }
                })
            }
        return max
    }
}
